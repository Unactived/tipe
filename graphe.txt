#!/usr/bin/env python3
# -*- coding: utf-8 -*-

import matplotlib.pyplot as plt
import numpy as np

a = np.linspace(0,1,100)

b = [-a[i]*np.log2(a[i]) for i in range(100)]
plt.plot(a, b, 'r', label="Gain d'information")

c = [a[i]*(1-a[i]) for i in range(100)]
plt.plot(a, c, 'b', label="Indice de Gini")

plt.legend()
plt.show()
